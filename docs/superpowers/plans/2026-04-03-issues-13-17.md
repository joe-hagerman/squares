### Steps

- [ ] **Step 1: Create ShareSheet.jsx**

Create `src/components/ShareSheet.jsx` with this content:

```jsx
import { useState } from 'react'
import QRCodeDisplay from './QRCodeDisplay'

/**
 * ShareSheet — bottom sheet with shareable URLs and QR codes.
 *
 * Props:
 *   boardId   – string UUID
 *   isLocked  – boolean; when true, only shows the live board link (no join link)
 *   onClose   – callback to dismiss the sheet
 */
export default function ShareSheet({ boardId, isLocked, onClose }) {
  const origin = window.location.origin
  const liveUrl = `${origin}/board/${boardId}`
  const joinUrl = `${origin}/board/${boardId}/join`

  return (
    <>
      {/* Backdrop */}
      <div
        onClick={onClose}
        style={{
          position: 'fixed', inset: 0, zIndex: 40,
          background: 'rgba(0,0,0,0.6)',
        }}
      />

      {/* Sheet */}
      <div
        style={{
          position: 'fixed', bottom: 0, left: 0, right: 0, zIndex: 50,
          background: 'var(--sq-surface)',
          borderTop: '1px solid rgba(var(--sq-accent-rgb),0.2)',
          borderRadius: '12px 12px 0 0',
          maxHeight: '85vh',
          overflowY: 'auto',
        }}
      >
        {/* Sheet header */}
        <div
          className="flex items-center justify-between px-5 py-4"
          style={{ borderBottom: '1px solid rgba(var(--sq-alpha),0.08)' }}
        >
          <span className="font-display text-sm font-bold tracking-[0.2em] uppercase text-white">Share</span>
          <button
            onClick={onClose}
            className="font-mono text-xs text-gray-400 hover:text-white transition-colors px-2 py-1"
          >
            ✕
          </button>
        </div>

        <div className="px-5 py-5 space-y-6">
          {/* Live board section — always shown */}
          <ShareSection
            label="Watch Live"
            description="Share the live scoreboard"
            url={liveUrl}
          />

          {/* Join section — only when board is open */}
          {!isLocked && (
            <ShareSection
              label="Join Board"
              description="Invite players to claim squares"
              url={joinUrl}
            />
          )}
        </div>
      </div>
    </>
  )
}

function ShareSection({ label, description, url }) {
  const [copied, setCopied] = useState(false)

  function copy() {
    navigator.clipboard.writeText(url)
    setCopied(true)
    setTimeout(() => setCopied(false), 2000)
  }

  return (
    <div className="space-y-3">
      {/* Section header */}
      <div className="flex items-center gap-2">
        <span className="w-0.5 h-3 rounded-full" style={{ background: 'var(--sq-accent)' }} />
        <span className="font-display text-[11px] font-semibold tracking-[0.2em] text-gray-400 uppercase">{label}</span>
      </div>
      <p className="font-mono text-[10px] text-gray-500 tracking-wider">{description}</p>

      {/* QR code */}
      <div className="flex justify-center py-2">
        <QRCodeDisplay url={url} label="" size={140} />
      </div>

      {/* URL + copy */}
      <div className="flex gap-2">
        <input
          readOnly
          value={url}
          className="flex-1 font-mono text-xs text-gray-400 truncate px-3 py-2 rounded-sm outline-none"
          style={{ background: 'rgba(var(--sq-alpha),0.04)', border: '1px solid rgba(var(--sq-alpha),0.07)' }}
        />
        <button
          onClick={copy}
          className="font-display text-xs font-bold tracking-wider uppercase px-3 py-2 rounded-sm transition-all whitespace-nowrap"
          style={{
            background: copied ? 'rgba(16,185,129,0.15)' : 'rgba(var(--sq-accent-rgb),0.12)',
            color: copied ? '#10b981' : '#f59e0b',
            border: `1px solid ${copied ? 'rgba(16,185,129,0.3)' : 'rgba(var(--sq-accent-rgb),0.2)'}`,
          }}
        >
          {copied ? 'Copied' : 'Copy'}
        </button>
      </div>
    </div>
  )
}
```
